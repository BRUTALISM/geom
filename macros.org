* Helper macros
** Matrix item calculation
#+BEGIN_SRC clojure :noweb-ref mat-ops  
  (defmacro mat-item
    ([a1 a2 b1 b2]
       `(+ (* ~a1 ~a2) (* ~b1 ~b2)))
    ([a1 a2 b1 b2 c1 c2]
       `(+ (+ (* ~a1 ~a2) (* ~b1 ~b2)) (* ~c1 ~c2)))
    ([a1 a2 b1 b2 c1 c2 d]
       `(+ (+ (+ (* ~a1 ~a2) (* ~b1 ~b2)) (* ~c1 ~c2)) ~d))
    ([a1 a2 b1 b2 c1 c2 d1 d2]
       `(+ (+ (+ (* ~a1 ~a2) (* ~b1 ~b2)) (* ~c1 ~c2)) (* ~d1 ~d2))))
  
  (defmacro det-item
    [a b c d, e f g h, i j k l, m n o p]
    `(+ (- (- (* (* ~a ~b) (* ~c ~d))
              (* (* ~e ~f) (* ~g ~h)))
           (* (* ~i ~j) (* ~k ~l)))
        (* (* ~m ~n) (* ~o ~p))))
#+END_SRC

    [voxels idx mask id]
    `(if (~voxels ~idx) ~id (bit-or ~id ~mask)))
